package com.traine.managment.service;

import java.time.LocalDateTime;
import java.util.List;

import org.modelmapper.ModelMapper;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.web.server.ResponseStatusException;

import com.traine.managment.dto.activityStoryLine.StoryLineCommentRequestDTO;
import com.traine.managment.dto.activityStoryLine.StoryLineCommentResponseDTO;
import com.traine.managment.dto.activityStoryLine.StoryLineGetResponseDTO;
import com.traine.managment.enums.activityStoryLine.ActivityStoryLineType;
import com.traine.managment.model.Activity;
import com.traine.managment.model.ActivityStoryLine;
import com.traine.managment.model.User;
import com.traine.managment.repository.ActivityRepository;
import com.traine.managment.repository.ActivityStoreLineRepository;
import com.traine.managment.repository.UserRepository;

import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.http.HttpStatus;

@Service
public class ActivityStoreLineService {

    @Autowired
    private ActivityStoreLineRepository storyLineRepository;

    @Autowired
    private AuthService authService;

    @Autowired
    private ActivityRepository activityRepository;

    @Autowired
    private ModelMapper modelMapper;

    @Autowired
    private UserRepository userRepository;

    public ActivityStoryLine createActivityStoreLine(Activity activity,
            ActivityStoryLineType type,
            String content) {

        User author = authService.getAuthenticatedUser();

        ActivityStoryLine line = ActivityStoryLine.builder()
                .activity(activity)
                .author(author)
                .type(type)
                .content(content)
                .createdAt(LocalDateTime.now())
                .build();

        return storyLineRepository.save(line);

    }

    private Activity verifyIfAnActivityExistsAndThenReturn(Long id) {
        return activityRepository.findById(id)
                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Atividade não encontrada"));
    }

    private User verifyAUserAndThenReturn(String email) {
        return userRepository.findByEmail(email)
                .orElseThrow(() -> new ResponseStatusException(HttpStatus.BAD_REQUEST,
                        "User não encontrado."));
    }

    private void verifyIfUserIsParticipantOf(User user, Activity activity) {

        boolean haveAuthority = activity.areParticipant(user);

        if (!haveAuthority) {
            throw new ResponseStatusException(HttpStatus.FORBIDDEN,
                    "Usuário não participa dessa atividade");
        }
    }

    private StoryLineGetResponseDTO transformStoryLineInto(ActivityStoryLine activityStoryLine) {
        return modelMapper.map(activityStoryLine, StoryLineGetResponseDTO.class);
    }

    public Page<StoryLineGetResponseDTO> getStoryLines(Long activityId, String email, Pageable pageable) {
        Activity activity = verifyIfAnActivityExistsAndThenReturn(activityId);

        User user = verifyAUserAndThenReturn(email);

        verifyIfUserIsParticipantOf(user, activity);

        Page<ActivityStoryLine> storyLines = storyLineRepository.findByActivityId(activityId, pageable);

        return storyLines.map(this::transformStoryLineInto);
    }

    public StoryLineCommentResponseDTO makeAComment(Long activityId, String email, StoryLineCommentRequestDTO body) {

        Activity activity = verifyIfAnActivityExistsAndThenReturn(activityId);

        User user = verifyAUserAndThenReturn(email);

        verifyIfUserIsParticipantOf(user, activity);
        ActivityStoryLine activityStoryLine = createActivityStoreLine(activity, ActivityStoryLineType.COMENTARIO,
                body.getComment());

        return StoryLineCommentResponseDTO.builder()
                .id(activityStoryLine.getId())
                .authorEmail(user.getEmail())
                .authorName(user.getName())
                .type(activityStoryLine.getType())
                .content(activityStoryLine.getContent())
                .createdAt(activityStoryLine.getCreatedAt())
                .build();
    }
}
