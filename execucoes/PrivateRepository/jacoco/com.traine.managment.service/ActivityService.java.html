        @Transactional
        private void createNotifications(Activity activity, String message, String subject, String title) {
                Set<User> participants = activity.getParticipants();

                participants.forEach(participant -> {
                        Notification notification = notificationService.createAnActivityNotification(participant,
                                        activity, message, subject, title);
                        participant.addNotification(notification);
                });

                userRepository.saveAll(participants);
        }

        private PostActivityResponseDTO mapToResponseCreate(Activity activity) {
                return new PostActivityResponseDTO(
                                activity.getId(),
                                activity.getDescription(),
                                activity.getActivityType(),
                                new HashSet<>(activity.getTrainees())
                                                .stream()
                                                .map(user -> modelMapper.map(user, CreateUserResponseDTO.class))
                                                .collect(Collectors.toSet()),

                                new HashSet<>(activity.getValidators())
                                                .stream()
                                                .map(user -> modelMapper.map(user, CreateUserResponseDTO.class))
                                                .collect(Collectors.toSet()),
                                activity.getTerm(),
                                activity.getActivityPriority(),
                                activity.getAuxiliaryFilesKeys());
        }

        private Activity create(PostActivityRequestDTO body) {
                Activity activity = new Activity();
                activity.setDescription(body.getDescription());
                activity.setActivityType(body.getActivityType());
                activity.setActivityPriority(body.getActivityPriority());
                activity.setTerm(body.getTerm());
                activity.setActivityStatus(ActivityStatus.PENDENTE);
                activity.setCreatedAt(LocalDate.now());

                Set<User> trainees = userRepository.findAllById(body.getTrainees())
                                .stream().collect(Collectors.toSet());
                Set<User> validators = userRepository.findAllById(body.getValidators())
                                .stream().collect(Collectors.toSet());

                activity.setTrainees(trainees);
                activity.setValidators(validators);

                activity.addAuxiliariesFileKeys(body.getAuxiliaryFilesKeys());

                activity = activityRepository.save(activity);

                return activity;
        }

        private void afterCreateOfTheActivity(Activity activity) {

                try {
                        insertIntoGoogleCalendar(activity);
                } catch (Exception e) {
                        e.printStackTrace();
                }

                try {
                        createNotifications(
                                        activity,
                                        "Uma nova atividade foi criada e você está associado a ela.",
                                        "Atividade criada: " + activity.getId(),
                                        "Uma nova atividade foi criada");
                } catch (Exception e) {
                        e.printStackTrace();
                }

                try {
                        insertIntoTrello(activity);
                } catch (Exception e) {
                        e.printStackTrace();
                }

                try {
                        activityStoreLineService.createActivityStoreLine(activity,
                                        ActivityStoryLineType.ATIVIDADE_CRIADA,
                                        activity.getCreatedAt().toString());

                } catch (Exception e) {
                        e.printStackTrace();
                }

        }

        public PostActivityResponseDTO createAnActivityInsertIntoGoogleCalendarAndSendNotification(
                        PostActivityRequestDTO body) {

                boolean allFilesExist = fileStorageService.allExists(
                                body.getAuxiliaryFilesKeys());

                if (!allFilesExist) {
                        throw new ResponseStatusException(
                                        HttpStatus.BAD_REQUEST,
                                        "Arquivo(s) inválido(s).");
                }

                Activity activity = create(body);

                afterCreateOfTheActivity(activity);

                return mapToResponseCreate(activity);

        }

        private Activity verifyIfAnActivityExistsAndThenReturn(Long id) {
                return activityRepository.findById(id)
                                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                                                "Atividade não encontrada"));
        }

        private User verifyAUserAndThenReturn(String email) {
                return userRepository.findByEmail(email)
                                .orElseThrow(() -> new ResponseStatusException(HttpStatus.BAD_REQUEST,
                                                "User não encontrado."));
        }

        private Activity put(Activity activity, PatchActivityRequestDTO body) {

                if (body.getDescription() != null) {
                        activity.setDescription(body.getDescription());
                }

                if (body.getActivityType() != null) {
                        activity.setActivityType(body.getActivityType());
                }

                if (body.getActivityPriority() != null) {
                        activity.setActivityPriority(body.getActivityPriority());
                }

                if (body.getTerm() != null) {
                        activity.setTerm(body.getTerm());
                }

                if (body.getTrainees() != null) {
                        Set<User> trainees = userRepository.findAllById(body.getTrainees())
                                        .stream().collect(Collectors.toSet());
                        activity.setTrainees(trainees);
                }

                if (body.getValidators() != null) {
                        Set<User> validators = userRepository.findAllById(body.getValidators())
                                        .stream().collect(Collectors.toSet());
                        activity.setValidators(validators);
                }

                activity = activityRepository.save(activity);

                return activity;
        }

        private void afterPutActivity(Activity activity, String diferences) {
                try {
                        googleService.updateGoogleCalendar(activity, userRepository.findUsersByActivity(activity));
                        createNotifications(activity, "Essa atividade foi modificada.",
                                        "Atividade modificada: " + activity.getId(),
                                        "Uma das suas atividades foi modificada");
                        trelloService.updateCard(activity, activity.getParticipants());
                } catch (Exception e) {

                }

                try {

                        activityStoreLineService.createActivityStoreLine(activity,
                                        ActivityStoryLineType.ALTERACAO_ATIVIDADE,
                                        "A atividade foi alterada: \n" + diferences);
                } catch (Exception e) {

                }

        }

        public PatchActivityResponseDTO putNoticationAndUpdateGoogleCalendarAndSendNotification(Long id,
                        PatchActivityRequestDTO body) {

                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                String differences = activity.differenceBetweenPutDto(body);

                activity = put(activity, body);

                afterPutActivity(activity, differences);

                return modelMapper.map(activity, PatchActivityResponseDTO.class);
        }

        @Transactional
        public void delete(Long id) {

                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                trelloService.deleteCard(activity.getId(), activity.getParticipants());
                googleService.deleteEventByActivity(activity);

                // TODO: Fazer com que notificações sejam enviadas
                trelloActivityCardRepository.deleteByActivityId(id);

                activityStoreLineRepository.deleteAllByActivityDirectly(activity);

                activityRepository.deleteById(id);

        }

        private GetActivityIParticipateResponseDTO transformActivityInto(Activity activity) {
                GetActivityIParticipateResponseDTO dto = new GetActivityIParticipateResponseDTO();
                dto.setId(activity.getId());
                dto.setDescription(activity.getDescription());
                dto.setActivityType(activity.getActivityType());
                dto.setActivityPriority(activity.getActivityPriority());
                dto.setActivityStatus(activity.getActivityStatus());
                dto.setTerm(activity.getTerm());
                dto.setTrainees(activity.getTrainees().stream()
                                .map(trainee -> modelMapper.map(trainee,
                                                CreateUserResponseDTO.class))
                                .collect(Collectors.toSet()));
                dto.setValidators(activity.getValidators().stream()
                                .map(validator -> modelMapper.map(validator,
                                                CreateUserResponseDTO.class))
                                .collect(Collectors.toSet()));

                return dto;
        }

        public Set<GetActivityIParticipateResponseDTO> getAll() {
                return activityRepository.findAll()
                                .stream()
                                .map(activity -> {
                                        return transformActivityInto(activity);
                                })
                                .collect(Collectors.toSet());
        }

        public Set<GetActivityIParticipateResponseDTO> getAllWichIParticipate(String email) {
                User user = verifyAUserAndThenReturn(email);

                List<Activity> activities = activityRepository
                                .findByTraineesContainsOrValidatorsContains(user, user);

                return activities.stream()
                                .map(activity -> {
                                        return transformActivityInto(activity);
                                })
                                .collect(Collectors.toSet());
        }

        private void verifyIfUserIsParticipantOf(User user, Activity activity) {

                boolean haveAuthority = activity.areParticipant(user);

                if (!haveAuthority) {
                        throw new ResponseStatusException(HttpStatus.FORBIDDEN,
                                        "Usuário não participa dessa atividade");
                }
        }

        private void verifyIfUserIsTraineeOf(User user, Activity activity) {
                boolean haveAuthority = activity.areTrainee(user);

                if (!haveAuthority) {
                        throw new ResponseStatusException(HttpStatus.FORBIDDEN,
                                        "Usuário não é estagiário dessa atividade.");
                }
        }

        public GetActivityResponseDTO getById(String email, Long id) {
                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                User user = verifyAUserAndThenReturn(email);

                verifyIfUserIsParticipantOf(user, activity); // Deveria permitir que gerentes acessacem isso.

                return modelMapper.map(activity, GetActivityResponseDTO.class);
        }

        private void afterPathInProgress(Activity activity) {
                try {

                        createNotifications(activity, "Atividade está em adamento",
                                        "Progresso na atividade: " + activity.getId(), "Progresso na atividade");
                        trelloService.moveCardTo(activity, activity.getParticipants(), ActivityStatus.EM_ANDAMENTO);

                } catch (Exception e) {
                        System.out.println(e.getStackTrace());
                }

                try {
                        activityStoreLineService.createActivityStoreLine(activity, ActivityStoryLineType.MUDANCA_ESTADO,
                                        "O status da atividade foi altarado para em andamento");
                } catch (Exception e) {
                        System.out.println(e.getStackTrace());

                }

        }

        private void verifyIfTheActivityCanGoToInProgress(Activity activity) {
                if (!activity.getActivityStatus().equals(ActivityStatus.PENDENTE)) {
                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST,
                                        "A atividade não está pendente para ser colocada em andamento.");
                }
        }

        private void changeTheStatus(Activity activity, ActivityStatus activityStatus) {
                activity.setActivityStatus(activityStatus);
                activity = activityRepository.save(activity);

        }

        public PathActivityProgressResponseDTO pathInProgress(String email, Long id) {

                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                User user = verifyAUserAndThenReturn(email);

                verifyIfUserIsTraineeOf(user, activity);

                verifyIfTheActivityCanGoToInProgress(activity);

                changeTheStatus(activity, ActivityStatus.EM_ANDAMENTO);

                afterPathInProgress(activity);

                return modelMapper.map(activity, PathActivityProgressResponseDTO.class);

        }

        private void afterPathAwaitEvaluation(Activity activity) {
                try {
                        createNotifications(activity,
                                        "Essa atividade está aguardado avaliação. Um dos validadores deve aceitá-la ou rejeitá-la",
                                        "Progresso na atividade: " + activity.getId(), "Progresso na atividade");
                        trelloService.moveCardTo(activity, activity.getParticipants(),
                                        ActivityStatus.AGUARDANDO_AVALIACAO);

                } catch (Exception e) {
                        System.out.println(e.getMessage());
                }

                try {
                        activityStoreLineService.createActivityStoreLine(activity, ActivityStoryLineType.MUDANCA_ESTADO,
                                        "O status da atividade foi alterado para aguardando avaliação");
                } catch (Exception e) {
                        System.out.println(e.getMessage());
                }
        }

        private void verifyIfTheActivityCanGoToAwaitEvaluation(Activity activity) {

                if (!(activity.getActivityStatus().equals(ActivityStatus.EM_ANDAMENTO)
                                || activity.getActivityStatus().equals(ActivityStatus.AJUSTE_NECESSARIO))) {
                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST,
                                        "A atividade não está em andamento para ser colocada em aguardando avaliação.");
                }
        }

        public PathActivityProgressResponseDTO pathAwaitEvaluation(String email, Long id) {
                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                User user = verifyAUserAndThenReturn(email);

                verifyIfUserIsTraineeOf(user, activity);

                verifyIfTheActivityCanGoToAwaitEvaluation(activity);

                changeTheStatus(activity, ActivityStatus.AGUARDANDO_AVALIACAO);

                afterPathAwaitEvaluation(activity);

                return modelMapper.map(activity, PathActivityProgressResponseDTO.class);
        }

        private void afterEvalute(Activity activity, boolean wasAccept) {
                try {

                        createNotifications(activity,
                                        wasAccept ? "Atividade Concluida"
                                                        : "A atividade foi rejeitada, ajuste são necessários",
                                        "Progresso na atividade: " + activity.getId(), "Progresso na atividade");

                } catch (Exception e) {
                        System.out.println(e.getMessage());
                }

                try {
                        if (wasAccept) {
                                trelloService.moveCardTo(activity, activity.getParticipants(),
                                                ActivityStatus.CONCLUIDO);
                        } else {

                                trelloService.moveCardTo(activity, activity.getParticipants(),
                                                ActivityStatus.AJUSTE_NECESSARIO);
                        }

                } catch (Exception e) {
                        System.out.println(e.getMessage());
                }

                try {
                        if (wasAccept) {
                                activityStoreLineService.createActivityStoreLine(activity,
                                                ActivityStoryLineType.MUDANCA_ESTADO,
                                                "O status da atividade foi alterado para concluida.");

                        } else {
                                activityStoreLineService.createActivityStoreLine(activity,
                                                ActivityStoryLineType.MUDANCA_ESTADO,
                                                "O status da atividade foi alterado para ajustes necessários.");

                        }

                } catch (Exception e) {
                        System.out.println(e.getMessage());
                }

        }

        private void haveAuthorityToEvaluate(User user, Activity activity) {
                boolean isValidator = activity.areValidator(user);

                if (!user.getRole().equals(Role.MANAGMENT) && !isValidator) {
                        throw new ResponseStatusException(HttpStatus.FORBIDDEN,
                                        "Usuário não tem permissão para alterar essa atividade.");
                }
        }

        private void verifyIfTheActivityCanGoToAjustmentOrFinished(Activity activity) {
                if (activity.getActivityStatus() != ActivityStatus.AGUARDANDO_AVALIACAO
                                || activity.getActivityStatus() == ActivityStatus.CONCLUIDO) {

                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST,
                                        "A atividade não pode já estar concluida ou em outro estado que não seja aguardando avaliação.");
                }
        }

        public PathActivityProgressResponseDTO evaluate(String email, Long id, PatchEvaluateRequestDTO body) {

                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                User user = verifyAUserAndThenReturn(email);

                haveAuthorityToEvaluate(user, activity);

                verifyIfTheActivityCanGoToAjustmentOrFinished(activity);

                if (body.isAccepted()) {
                        changeTheStatus(activity, ActivityStatus.CONCLUIDO);
                        activity.finishedAt();

                } else {
                        changeTheStatus(activity, ActivityStatus.AJUSTE_NECESSARIO);
                }

                afterEvalute(activity, body.isAccepted());

                return modelMapper.map(activity, PathActivityProgressResponseDTO.class);
        }

        private void haveAuthorizationToModifyAuxFiles(User user, Activity activity) {
                boolean haveAutorization = user.getRole().equals(Role.MANAGMENT) || activity.areValidator(user);

                if (!haveAutorization) {
                        throw new ResponseStatusException(HttpStatus.FORBIDDEN,
                                        "Você não tem autorização para modificar os arquivos auxiliares dessa atividade");
                }
        }

        private void verifyIfTheFileKeysAreValids(Set<String> auxKeys) {
                if (!fileStorageService.allExists(auxKeys)) {
                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST,
                                        "Arquivo(s) inválido(s)");
                }
        }

        public AddNewAuxFileResponse addAuxFiles(Long id, String email, AddNewAuxFileRequest body) {

                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                User user = verifyAUserAndThenReturn(email);

                haveAuthorizationToModifyAuxFiles(user, activity);

                verifyIfTheFileKeysAreValids(body.getAuxiliaryFilesKeys());

                activity.addAuxiliariesFileKeys(body.getAuxiliaryFilesKeys());

                activityRepository.save(activity);

                afterAddAuxFiles(activity);

                return new AddNewAuxFileResponse(activity.getId(),
                                activity.getAuxiliaryFilesKeys());

        }

        private void afterDeleteAuxFile(Activity activity) {
                createNotifications(activity, "Um novo material de apoio foi adicionado a essa atividade.",
                                "Material de apoio adicionado na atividade " + activity.getId(),
                                "Material de apoio adicionado");

                try {
                        activityStoreLineService.createActivityStoreLine(activity,
                                        ActivityStoryLineType.ARQUIVOS_AUXILIARES_REMOVIDOS,
                                        "Arquivo(s) auxiliares foram removidos.");
                } catch (Exception e) {

                }
        }

        public void deleteAuxFile(Long id, String fileKey, String email) {
                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                User user = verifyAUserAndThenReturn(email);

                haveAuthorizationToModifyAuxFiles(user, activity);

                if (!activity.haveAuxiliarKey(fileKey)) {
                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST,
                                        "File key não existente.");

                }

                activity.deleteAuxiliarFileKey(fileKey);

                activityRepository.save(activity);

                afterDeleteAuxFile(activity);
        }

        private void afterAddAnswerFiles(Activity activity) {
                activityStoreLineService.createActivityStoreLine(activity,
                                ActivityStoryLineType.ARQUIVO_RESPOSTA_ADICIONADO,
                                "Arquivo(s) de resposta foram adicionados.");
        }

        private void haveAuthorizationToModifyAnswerFiles(User user, Activity activity) {
                boolean haveAutorization = activity.areTrainee(user);

                if (!haveAutorization) {
                        throw new ResponseStatusException(HttpStatus.FORBIDDEN,
                                        "Você não tem autorização para adicionar arquivos a essa atividade");
                }
        }

        private void verifyIfTheActivityIsInTheAllowedStatusToModifyAnswerFiles(Activity activity) {
                var allowedStatus = List.of(ActivityStatus.EM_ANDAMENTO, ActivityStatus.AJUSTE_NECESSARIO);

                if (!allowedStatus.contains(activity.getActivityStatus())) {
                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Status inválido para esta ação.");
                }
        }

        public AddNewAnswerFileResponse addAnswerFiles(Long id, String email, AddNewAnswerFileRequest body) {
                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                User user = verifyAUserAndThenReturn(email);

                haveAuthorizationToModifyAnswerFiles(user, activity);

                verifyIfTheFileKeysAreValids(body.getAnswerFilesKeys());

                verifyIfTheActivityIsInTheAllowedStatusToModifyAnswerFiles(activity);

                activity.addAnswerFileKeys(body.getAnswerFilesKeys());
                activityRepository.save(activity);

                afterAddAnswerFiles(activity);

                return new AddNewAnswerFileResponse(id, body.getAnswerFilesKeys());
        }

        private void afterDeleteAnswerFile(Activity activity) {

                try {
                        activityStoreLineService.createActivityStoreLine(activity,
                                        ActivityStoryLineType.ARQUIVO_RESPOSTA_REMOVIDO,
                                        "Arquivo(s) de resposta foram removidos.");
                } catch (Exception e) {

                }

        }

        public void deleteAnswerFile(Long id, String fileKey, String email) {
                Activity activity = verifyIfAnActivityExistsAndThenReturn(id);

                User user = verifyAUserAndThenReturn(email);

                haveAuthorizationToModifyAnswerFiles(user, activity);

                if (!activity.haveAnswerKey(fileKey)) {
                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST,
                                        "File key não existente.");

                }

                verifyIfTheActivityIsInTheAllowedStatusToModifyAnswerFiles(activity);

                activity.deleteAnswerFileKey(fileKey);

                activityRepository.save(activity);

                afterDeleteAnswerFile(activity);
        }
