package com.traine.managment.service;

import java.sql.Date;
import java.time.LocalDate;
import java.util.Arrays;
import java.util.List;

import org.modelmapper.ModelMapper;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;
import org.springframework.web.server.ResponseStatusException;

import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.domain.Sort;
import org.springframework.data.repository.query.Param;
import org.springframework.data.domain.Pageable;

import com.traine.managment.dto.activity.GetRecentActivityResponseDTO;
import com.traine.managment.dto.user.GetTraineeActivitiesSumaryResponseDTO;
import com.traine.managment.dto.user.GetTraineeActivitiesYearDTO;
import com.traine.managment.dto.user.GetValidatorsActivitiesSumaryResponseDTO;
import com.traine.managment.dto.user.ChangePasswordRequestDTO;
import com.traine.managment.dto.user.CreateUserRequestDTO;
import com.traine.managment.dto.user.CreateUserResponseDTO;
import com.traine.managment.dto.user.GetSafeIdentificationDTO;
import com.traine.managment.dto.user.MeUserResponseDTO;
import com.traine.managment.dto.user.PutUserRequestDTO;
import com.traine.managment.dto.user.PutUserResponseDTO;
import com.traine.managment.dto.user.UserGetResponseDTO;
import com.traine.managment.enums.activity.ActivityStatus;
import com.traine.managment.enums.user.Role;
import com.traine.managment.exceptions.EmailAlreadyExistsException;
import com.traine.managment.model.Notification;
import com.traine.managment.model.User;
import com.traine.managment.repository.ActivityRepository;
import com.traine.managment.repository.UserRepository;

@Service
public class UserService {

        @Autowired
        private UserRepository repository;

        @Autowired
        private ActivityRepository activityRepository;

        @Autowired
        private PasswordEncoder passwordEncoder;

        @Autowired
        private NotificationService notificationService;

        @Autowired
        private ModelMapper modelMapper;

        private void createNotification(User user, String message, String subject, String title) {
                Notification notification = notificationService.createUserNotification(user,
                                message,
                                subject, title);
                user.addNotification(notification);
        }

        public CreateUserResponseDTO createUser(CreateUserRequestDTO request) {

                if (repository.existsByEmail(request.getEmail())) {
                        throw new EmailAlreadyExistsException(request.getEmail());
                }

                User user = modelMapper.map(request, User.class);

                user.setPassword(passwordEncoder.encode(request.getPassword()));

                user = repository.save(user);

                createNotification(user,
                                "Seu usuário foi criado, você já pode logar com seu email e a senha que lhe foi dado.",
                                "Criação de usuário", "Usuário criado.");

                CreateUserResponseDTO response = modelMapper.map(user, CreateUserResponseDTO.class);

                return response;
        }

        public List<UserGetResponseDTO> getAllTrainees() {

                List<User> trainees = repository
                                .findByRole(Role.TRAINEE)
                                .orElseThrow(() -> new ResponseStatusException(
                                                HttpStatus.NOT_FOUND, "Nenhum trainee encontrado"));

                List<UserGetResponseDTO> response = trainees.stream()
                                .map(t -> modelMapper.map(t, UserGetResponseDTO.class))
                                .toList();

                return response;

        }

        public List<UserGetResponseDTO> getAllValidators() {

                List<User> validators = repository
                                .findByRoleIn(List.of(Role.ACTIVITY_MAKER, Role.MANAGMENT))
                                .orElseThrow(() -> new ResponseStatusException(
                                                HttpStatus.NOT_FOUND, "Nenhum validador encontrado"));

                List<UserGetResponseDTO> response = validators.stream()
                                .map(t -> modelMapper.map(t, UserGetResponseDTO.class))
                                .toList();

                return response;

        }

        public PutUserResponseDTO updateUser(PutUserRequestDTO request, String email) {
                User user = repository.findByEmail(email)
                                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                                                "Usuário não encontrado."));

                user.setName(request.getName());
                user.setEmail(request.getEmail());
                user.setSector(request.getSector());

                user = repository.save(user);

                createNotification(user, "Seu usuário foi alterado.", "Alteração de cadastro", "Usuário Atualizado.");

                PutUserResponseDTO response = modelMapper.map(user, PutUserResponseDTO.class);
                return response;
        }

        public void changePassword(ChangePasswordRequestDTO request, String email) {
                User user = repository.findByEmail(email)
                                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                                                "Usuário não encontrado."));

                if (!passwordEncoder.matches(request.getCurrentPassword(), user.getPassword())) {
                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Senha atual incorreta.");
                }

                if (passwordEncoder.matches(request.getNewPassword(), user.getPassword())) {
                        throw new ResponseStatusException(HttpStatus.BAD_REQUEST,
                                        "A nova senha deve ser diferente da senha atual.");
                }

                user.setPassword(passwordEncoder.encode(request.getNewPassword()));

                createNotification(user, "Seu usuário foi alterado.", "Alteração de senha",
                                "Sua senha foi alterada com sucesso.");

                repository.save(user);

        }

        private Long[] mapToMonthlyCounts(List<Object[]> raw) {
                Long[] result = new Long[12];
                Arrays.fill(result, 0L);

                raw
                                .stream()
                                .forEach(o -> {
                                        int month = ((Number) o[0]).intValue() - 1;
                                        result[month] = (Long) o[1];
                                });
                return result;
        }

        public GetTraineeActivitiesSumaryResponseDTO getTraineeActivitiesSummary(String email) {
                User user = repository.findByEmail(email)
                                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                                                "Usuário não encontrado."));

                Long activitiesDone = activityRepository.countByUserAndStatus(
                                user, ActivityStatus.CONCLUIDO);
                Long activitiesPending = activityRepository.countByUserAndStatus(
                                user,
                                ActivityStatus.PENDENTE);

                Long[] activitiesParticipatedThisYear = mapToMonthlyCounts(
                                activityRepository.countFinishedByMonth(user.getId(),
                                                ActivityStatus.CONCLUIDO,
                                                LocalDate.now().getYear()));

                GetTraineeActivitiesSumaryResponseDTO response = new GetTraineeActivitiesSumaryResponseDTO();
                response.setActivitiesDone(activitiesDone);
                response.setActivitiesPending(activitiesPending);
                response.setActivitiesFinishedThisYear(activitiesParticipatedThisYear);

                return response;
        }

        public GetValidatorsActivitiesSumaryResponseDTO getValidatorsActivitiesSummary(String email) {
                User user = repository.findByEmail(email)
                                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                                                "Usuário não encontrado."));

                Long activitiesDone = activityRepository.countByUserAndStatus(
                                user, ActivityStatus.CONCLUIDO);
                Long activitiesPending = activityRepository.countByUserAndStatus(
                                user,
                                ActivityStatus.PENDENTE);
                Long activitiesInAdjustment = activityRepository.countByUserAndStatus(
                                user,
                                ActivityStatus.AJUSTE_NECESSARIO);
                Long activitiesInProgress = activityRepository.countByUserAndStatus(
                                user,
                                ActivityStatus.EM_ANDAMENTO);

                Long[] activitiesParticipatedThisYear = mapToMonthlyCounts(activityRepository.countParticipatedByYear(
                                user.getId(),
                                LocalDate.now().getYear()));

                GetValidatorsActivitiesSumaryResponseDTO response = new GetValidatorsActivitiesSumaryResponseDTO();
                response.setActivitiesDone(activitiesDone);
                response.setActivitiesPending(activitiesPending);
                response.setActivitiesInAdjustment(activitiesInAdjustment);
                response.setActivitiesInProgress(activitiesInProgress);
                response.setActivitiesParticipatedThisYear(activitiesParticipatedThisYear);

                return response;
        }

        public List<GetRecentActivityResponseDTO> getRecentActivities(String email) {
                User user = repository.findByEmail(email)
                                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                                                "Usuário não encontrado."));

                Pageable pageable = PageRequest.of(0, 5, Sort.by("term").descending());

                List<GetRecentActivityResponseDTO> response = activityRepository
                                .findByUserAndNotFinishedOrderByTermDesc(user.getId(), ActivityStatus.CONCLUIDO,
                                                pageable)
                                .stream()
                                .map(a -> modelMapper.map(a, GetRecentActivityResponseDTO.class))
                                .toList();

                return response;
        }

        public GetTraineeActivitiesYearDTO[] getTraineeActivitiesYearQuantity() {
                List<User> trainees = repository
                                .findByRole(Role.TRAINEE)
                                .orElseThrow(() -> new ResponseStatusException(
                                                HttpStatus.NOT_FOUND, "Nenhum trainee encontrado"));

                GetTraineeActivitiesYearDTO[] response = new GetTraineeActivitiesYearDTO[trainees.size()];

                response = trainees.stream()
                                .map(t -> {

                                        Long[] activitiesParticipatedThisYear = mapToMonthlyCounts(activityRepository
                                                        .countParticipatedTraineeByMonthForYear(t.getId(),
                                                                        java.time.LocalDate.now().getYear()));

                                        return new GetTraineeActivitiesYearDTO(t.getName(),
                                                        activitiesParticipatedThisYear);
                                })
                                .toArray(GetTraineeActivitiesYearDTO[]::new);

                return response;
        }

        public GetSafeIdentificationDTO getSafeIdentication(String email) {

                User user = repository.findByEmail(email).orElseThrow(() -> new ResponseStatusException(
                                HttpStatus.NOT_FOUND, "Usuário não encontrado."));

                return modelMapper.map(user, GetSafeIdentificationDTO.class);
        }
}
