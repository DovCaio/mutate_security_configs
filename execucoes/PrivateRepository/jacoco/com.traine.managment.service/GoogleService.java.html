package com.traine.managment.service;

import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.HashMap;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.HttpEntity;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpMethod;
import org.springframework.http.HttpStatus;
import org.springframework.http.HttpStatusCode;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;
import org.springframework.util.LinkedMultiValueMap;
import org.springframework.util.MultiValueMap;
import org.springframework.web.client.HttpClientErrorException;
import org.springframework.web.client.RestTemplate;
import org.springframework.web.server.ResponseStatusException;

import com.traine.managment.model.Activity;
import com.traine.managment.model.GoogleCalendarCredentials;
import com.traine.managment.model.GoogleCalendarEvent;
import com.traine.managment.model.User;
import com.traine.managment.repository.GoogleCredentialsRepository;
import com.traine.managment.repository.UserRepository;

import jakarta.transaction.Transactional;

@Service
public class GoogleService {
    @Value("${app.client-id}")
    private String clientId;

    @Value("${app.client-secret}")
    private String clientSecret;

    @Autowired
    private GoogleCredentialsRepository googleCaledarCredentialsRepository;
    @Autowired
    private UserRepository userRepository;
    @Autowired
    private RequestsService requestsService;
    @Autowired
    private GoogleCalendarEventService googleCalendarEventService;

    private final String url = "https://oauth2.googleapis.com/token";

    private void save(GoogleCalendarCredentials credentials, User user) {

        googleCaledarCredentialsRepository.save(credentials);
        userRepository.save(user);

    }

    private MultiValueMap<String, String> createBodyForTokensRequest(String code) {

        MultiValueMap<String, String> body = new LinkedMultiValueMap<>();
        body.add("code", code);
        body.add("client_id", clientId);
        body.add("client_secret", clientSecret);
        body.add("redirect_uri", "https://api.findmyride.site/google/callback"); // era bom ser uma variável de ambiente
        body.add("grant_type", "authorization_code");
        return body;

    }

    private Map<String, Object> makeRequest(String code) {

        MultiValueMap<String, String> body = createBodyForTokensRequest(code);

        HttpHeaders headers = requestsService.createHeaders(MediaType.APPLICATION_FORM_URLENCODED);

        return requestsService.makeForm(headers, body, url);
    }

    private GoogleCalendarCredentials turnIntoCredentials(String code, User user) {

        Map<String, Object> responseBody = makeRequest(code);

        if (responseBody == null) {
            throw new ResponseStatusException(
                    HttpStatus.BAD_GATEWAY,
                    "Erro ao comunicar com o Google");
        }

        GoogleCalendarCredentials credentials = new GoogleCalendarCredentials();
        credentials.setAccessToken((String) responseBody.get("access_token"));
        credentials.setRefreshToken((String) responseBody.get("refresh_token"));
        credentials.setCreateAt(LocalDateTime.now());
        user.setGoogleCredentials(credentials);

        return credentials;
    }

    public void turnCodeIntoCredentialsAndSave(String code, String state) {
        User user = userRepository.findBySafeIdentification(state)
                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Usuário não encontrado."));

        if (areConnected(user.getEmail()))
            throw new ResponseStatusException(HttpStatus.CONFLICT, "O usuário já está conectado");

        GoogleCalendarCredentials credentials = turnIntoCredentials(code, user);
        save(credentials, user);

    }

    public Boolean areConnected(String email) {
        User user = userRepository.findByEmail(email)
                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Usuário não encontrado."));

        return googleCaledarCredentialsRepository.existsByOwner(user);

    }

    private String refreshAccessToken(String refreshToken) {
        String url = "https://oauth2.googleapis.com/token";
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_FORM_URLENCODED);

        MultiValueMap<String, String> body = new LinkedMultiValueMap<>();
        body.add("client_id", clientId);
        body.add("client_secret", clientSecret);
        body.add("refresh_token", refreshToken);
        body.add("grant_type", "refresh_token");

        Map<String, Object> response = requestsService.makeForm(headers, body, url);

        Object accessToken = response.get("access_token");

        if (accessToken == null) {
            throw new RuntimeException("Erro ao renovar token: " + response);
        }

        return (String) accessToken;
    }

    private HttpHeaders createHeadersForInsertIntoCalendar(String accessToken) {
        HttpHeaders headers = new HttpHeaders();
        headers.setBearerAuth(accessToken);
        headers.setContentType(MediaType.APPLICATION_JSON);
        return headers;
    }

    private Map<String, Object> createBodyForInsertIntoGoogleCalendar(Activity activity) {

        Map<String, Object> body = new HashMap<>();

        body.put("summary", activity.getActivityType().getDisplayName());
        body.put("description", activity.getDescription());

        Map<String, String> start = new HashMap<>();
        start.put("date", activity.getTerm().toString());

        Map<String, String> end = new HashMap<>();
        end.put("date", activity.getTerm().plusDays(1).toString()); // Como não usa horário, e é um dia inteiro, tem que
                                                                    // ser assim.

        body.put("start", start);
        body.put("end", end);

        return body;

    }

    private Map<String, Object> makeGoogleRequestWithRetry(
            String url,
            HttpMethod method,
            HttpHeaders headers,
            Map<String, Object> body,
            GoogleCalendarCredentials credentials) {
        try {
            return requestsService.make(method, headers, body, url);

        } catch (HttpClientErrorException.Unauthorized e) {
            String newAccessToken = refreshAccessToken(credentials.getRefreshToken());

            credentials.setAccessToken(newAccessToken);
            googleCaledarCredentialsRepository.save(credentials);

            headers = createHeadersForInsertIntoCalendar(credentials.getAccessToken());

            return requestsService.make(method, headers, body, url);
        }
    }

    private String createEvent(GoogleCalendarCredentials credentials, Activity activity) {

        String url = "https://www.googleapis.com/calendar/v3/calendars/primary/events";

        Map<String, Object> body = createBodyForInsertIntoGoogleCalendar(activity);
        HttpHeaders headers = createHeadersForInsertIntoCalendar(credentials.getAccessToken());

        Map<String, Object> response = makeGoogleRequestWithRetry(url, HttpMethod.POST, headers, body, credentials);

        return (String) response.get("id");
    }

    @Async
    public void insertIntoGoogleCalendar(Activity activity, Set<User> participants) {
        Set<User> participantsConnected = participants.stream()
                .filter(participant -> areConnected(participant.getEmail())).collect(Collectors.toSet());

        participantsConnected.forEach(connected -> { // Isso daqui gera muitas requisições, porém entendo que seja mais
                                                     // fácil de dar manuntenção
            GoogleCalendarCredentials credential = googleCaledarCredentialsRepository.getByOwner(connected)
                    .orElseThrow();

            Optional<GoogleCalendarEvent> existing = googleCalendarEventService.getGoogleCalendarOptional(connected,
                    activity);

            String googleCalendarId = createEvent(credential, activity);

            if (!existing.isPresent()) {
                GoogleCalendarEvent googleCalendarEvent = GoogleCalendarEvent.builder()
                        .eventId(googleCalendarId)
                        .activity(activity)
                        .user(connected)
                        .build();
                googleCalendarEventService.create(googleCalendarEvent);
            }

        });
    }

    private void updateEvent(
            GoogleCalendarCredentials credentials,
            Activity activity,
            String eventId) {

        String url = "https://www.googleapis.com/calendar/v3/calendars/primary/events/" + eventId;

        Map<String, Object> body = createBodyForInsertIntoGoogleCalendar(activity);
        HttpHeaders headers = createHeadersForInsertIntoCalendar(credentials.getAccessToken());

        makeGoogleRequestWithRetry(url, HttpMethod.PUT, headers, body, credentials);
    }

    @Async
    public void updateGoogleCalendar(Activity activity, Set<User> participants) {

        Set<User> participantsConnected = participants.stream()
                .filter(participant -> areConnected(participant.getEmail())).collect(Collectors.toSet());

        participantsConnected.forEach(connected -> {

            GoogleCalendarCredentials credential = googleCaledarCredentialsRepository.getByOwner(connected)
                    .orElseThrow();

            Optional<GoogleCalendarEvent> event = googleCalendarEventService.getGoogleCalendarOptional(connected,
                    activity);

            if (event.isPresent()) {
                updateEvent(credential, activity, event.get().getEventId());
            } else {
                String googleCalendarId = createEvent(credential, activity);
                GoogleCalendarEvent googleCalendarEvent = GoogleCalendarEvent.builder()
                        .eventId(googleCalendarId)
                        .activity(activity)
                        .user(connected)
                        .build();
                googleCalendarEventService.create(googleCalendarEvent);
            }

        });

    }

    @Transactional
    public void deleteEventByActivity(Activity activity) {

        googleCalendarEventService.deleteAllByActivity(activity);

    }

}
