package com.traine.managment.service;

import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpMethod;
import org.springframework.http.HttpStatus;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;
import org.springframework.web.client.HttpClientErrorException;
import org.springframework.web.server.ResponseStatusException;
import org.springframework.web.util.UriComponentsBuilder;

import com.traine.managment.enums.activity.ActivityStatus;
import com.traine.managment.model.Activity;
import com.traine.managment.model.TrelloActivityCard;
import com.traine.managment.model.TrelloIntegration;
import com.traine.managment.model.User;
import com.traine.managment.repository.TrelloActivityCardRepository;
import com.traine.managment.repository.TrelloIntegrationsRepository;
import com.traine.managment.repository.UserRepository;

import jakarta.transaction.Transactional;

@Service
public class TrelloService {

        @Autowired
        private TrelloIntegrationsRepository trelloRepository;
        @Autowired
        private UserRepository userRepository;

        @Value("${trello.key}")
        private String trelloKey;

        @Autowired
        private RequestsService requestsService;

        @Autowired
        private TrelloActivityCardRepository trelloActivityCardRepository;

        private String createList(
                        String boardId,
                        String name,
                        String token) {

                String url = UriComponentsBuilder
                                .fromUriString("https://api.trello.com/1/lists")
                                .queryParam("name", name)
                                .queryParam("idBoard", boardId)
                                .queryParam("key", trelloKey)
                                .queryParam("token", token)
                                .build(false)
                                .toUriString();

                Map<String, Object> response = requestsService.make(
                                HttpMethod.POST,
                                new HttpHeaders(),
                                null,
                                url);

                Object id = response.get("id");

                if (id == null) {
                        System.out.println(
                                        "Erro ao criar lista no Trello.");
                }

                return id.toString();
        }

        private Map<ActivityStatus, String> createLists(String boardId, String token) {

                return Arrays.stream(ActivityStatus.values())
                                .collect(Collectors.toMap(
                                                status -> status,
                                                status -> createList(
                                                                boardId,
                                                                status.getDisplayName(),
                                                                token)));
        }

        private String createBoard(String token) {
                String url = UriComponentsBuilder
                                .fromUriString("https://api.trello.com/1/boards")
                                .queryParam("name", "Trainee Manager")
                                .queryParam("key", trelloKey)
                                .queryParam("token", token)
                                .build(false)
                                .toUriString();

                Map<String, Object> response = requestsService.make(
                                HttpMethod.POST,
                                new HttpHeaders(),
                                null,
                                url);

                return response.get("id").toString();

        }

        @Transactional
        public void createAnAssociations(String userEmail, String token) {

                User owner = userRepository.findByEmail(userEmail).orElseThrow(() -> new ResponseStatusException(
                                HttpStatus.NOT_FOUND, "Usuário não encontrado."));

                TrelloIntegration trelloIntegration = TrelloIntegration.builder()
                                .token(token)
                                .build();
                owner.setTrelloIntegration(trelloIntegration);

                String boardId = createBoard(token);

                trelloIntegration.setBoardId(boardId);

                trelloIntegration.setListsIds(createLists(boardId, token));

                userRepository.save(owner);

        }

        public Boolean isAssociated(String email) {
                User user = userRepository.findByEmail(email).orElseThrow(() -> new ResponseStatusException(
                                HttpStatus.NOT_FOUND, "Usuário não encontrado."));

                return trelloRepository.existsByUser(user);
        }

        private String createCard(
                        String listId,
                        String name,
                        String description,
                        String token) {

                String url = UriComponentsBuilder
                                .fromUriString(
                                                "https://api.trello.com/1/cards")
                                .queryParam("idList", listId)
                                .queryParam("name", name)
                                .queryParam("desc", description)
                                .queryParam("key", trelloKey)
                                .queryParam("token", token)
                                .build(false)
                                .toUriString();

                Map<String, Object> response = requestsService.make(
                                HttpMethod.POST,
                                new HttpHeaders(),
                                null,
                                url);

                Object id = response.get("id");

                if (id == null) {
                        throw new RuntimeException(
                                        "Erro ao criar card no Trello.");
                }

                return id.toString();
        }

        @Async
        @Transactional
        public void insertIntoTrello(Activity activity, Set<User> participants) {

                Set<User> associateds = participants.stream()
                                .filter(participant -> isAssociated(participant.getEmail()))
                                .collect(Collectors.toSet());

                associateds.forEach(associated -> {
                        TrelloIntegration trelloIntegration = trelloRepository.findByUser(associated).orElseThrow();
                        String pendentListId = trelloIntegration.getListsIds().get(ActivityStatus.PENDENTE);
                        String cardName = activity.getActivityType().getDisplayName();
                        String cardDescription = """
                                        Criada em: %s

                                        Termina em: %s

                                        Descrição:
                                        %s
                                        """
                                        .formatted(
                                                        activity.getCreatedAt(),
                                                        activity.getTerm(),
                                                        activity.getDescription());
                        String cardId = createCard(pendentListId, cardName, cardDescription,
                                        trelloIntegration.getToken());
                        TrelloActivityCard trelloActivityCard = TrelloActivityCard.builder()
                                        .activity(activity)
                                        .user(associated)
                                        .cardId(cardId)
                                        .build();
                        trelloActivityCardRepository.save(trelloActivityCard);

                });
        }

        private void moveCard(
                        String cardId,
                        String listId,
                        String token) {

                String url = UriComponentsBuilder
                                .fromUriString("https://api.trello.com/1/cards/" + cardId)
                                .queryParam("idList", listId)
                                .queryParam("key", trelloKey)
                                .queryParam("token", token)
                                .toUriString();

                requestsService.make(
                                HttpMethod.PUT,
                                new HttpHeaders(),
                                null,
                                url);
        }

        @Async
        @Transactional
        public void moveCardTo(Activity activity, Set<User> participants, ActivityStatus activityStatus) {

                Set<User> associateds = participants.stream()
                                .filter(participant -> isAssociated(participant.getEmail()))
                                .collect(Collectors.toSet());

                associateds.forEach(associated -> {
                        TrelloActivityCard trelloActivityCard = trelloActivityCardRepository
                                        .findByUserAndActivity(associated, activity).orElseThrow();
                        TrelloIntegration trelloIntegration = trelloRepository.findByUser(associated).orElseThrow();
                        String activityStausListId = trelloIntegration.getListsIds().get(activityStatus);
                        moveCard(trelloActivityCard.getCardId(), activityStausListId, trelloIntegration.getToken());
                });

        }

        private void updateCard(
                        String cardId,
                        String name,
                        String description,
                        String token) {

                String url = UriComponentsBuilder
                                .fromUriString("https://api.trello.com/1/cards/" + cardId)
                                .queryParam("name", name)
                                .queryParam("desc", description)
                                .queryParam("key", trelloKey)
                                .queryParam("token", token)
                                .build(false)
                                .toUriString();

                requestsService.make(
                                HttpMethod.PUT,
                                new HttpHeaders(),
                                null,
                                url);
        }

        @Async
        @Transactional
        public void updateCard(Activity activity, Set<User> participants) {

                Set<User> associateds = participants.stream()
                                .filter(participant -> isAssociated(participant.getEmail()))
                                .collect(Collectors.toSet());

                associateds.forEach(associated -> {
                        TrelloActivityCard trelloActivityCard = trelloActivityCardRepository
                                        .findByUserAndActivity(associated, activity).orElseThrow();
                        TrelloIntegration trelloIntegration = trelloRepository.findByUser(associated).orElseThrow();
                        String cardName = activity.getActivityType().getDisplayName();
                        String cardDescription = """
                                        Criada em: %s

                                        Termina em: %s

                                        Descrição:
                                        %s
                                        """
                                        .formatted(
                                                        activity.getCreatedAt(),
                                                        activity.getTerm(),
                                                        activity.getDescription());

                        updateCard(trelloActivityCard.getCardId(), cardName, cardDescription,
                                        trelloIntegration.getToken());
                });
        }

        private void deleteCard(
                        String cardId,
                        String token) {

                String url = UriComponentsBuilder
                                .fromUriString("https://api.trello.com/1/cards/" + cardId)
                                .queryParam("key", trelloKey)
                                .queryParam("token", token)
                                .toUriString();

                requestsService.make(
                                HttpMethod.DELETE,
                                new HttpHeaders(),
                                null,
                                url);
        }

        public void deleteCard(Long activityId, Set<User> participants) {

                Set<User> associateds = participants.stream()
                                .filter(p -> isAssociated(p.getEmail()))
                                .collect(Collectors.toSet());

                associateds.forEach(user -> {

                        trelloActivityCardRepository
                                        .findByUserIdAndActivityId(user.getId(), activityId)
                                        .ifPresent(card -> {

                                                TrelloIntegration integration = trelloRepository
                                                                .findByUser(user)
                                                                .orElse(null);

                                                try {
                                                        if (integration != null) {
                                                                deleteCard(card.getCardId(), integration.getToken());
                                                        }
                                                } catch (Exception e) {
                                                        System.out.println("Falha ao deletar no Trello: "
                                                                        + e.getMessage());
                                                }

                                                trelloActivityCardRepository.delete(card);
                                        });
                });
        }
}
