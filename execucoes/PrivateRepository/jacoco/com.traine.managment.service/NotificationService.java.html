package com.traine.managment.service;

import java.time.LocalDateTime;
import java.util.List;

import org.modelmapper.ModelMapper;
import org.springframework.data.domain.Pageable;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.crossstore.ChangeSetPersister.NotFoundException;
import org.springframework.data.domain.Page;
import org.springframework.http.HttpStatus;
import org.springframework.stereotype.Service;
import org.springframework.web.server.ResponseStatusException;

import com.traine.managment.dto.notification.GetHaveNotificationNotReadResponseDTO;
import com.traine.managment.dto.notification.GetNotificationResponseDTO;
import com.traine.managment.dto.notification.PathNotificationResponseDTO;
import com.traine.managment.enums.activity.ActivityType;
import com.traine.managment.enums.notification.NotificationType;
import com.traine.managment.interfaces.notification.EmailSender;
import com.traine.managment.model.Activity;
import com.traine.managment.model.Notification;
import com.traine.managment.model.User;
import com.traine.managment.repository.NotificationRepository;
import com.traine.managment.repository.UserRepository;

@Service
public class NotificationService {

    private final EmailSender emailSender;
    private final EmailTemplateService emailTemplateService;
    private final NotificationRepository notificationRepository;
    private final UserRepository userRepository;

    @Autowired
    private ModelMapper modelMapper;

    public NotificationService(
            EmailSender emailSender,
            EmailTemplateService emailTemplateService,
            NotificationRepository notificationRepository,
            UserRepository userRepository) {
        this.emailSender = emailSender;
        this.emailTemplateService = emailTemplateService;
        this.notificationRepository = notificationRepository;
        this.userRepository = userRepository;
    }

    public Notification createAnActivityNotification(User user, Activity activity, String message, String subject,
            String title) {
        Notification notification = Notification.builder()
                .message(message)
                .user(user)
                .recipient(user.getEmail())
                .title(title)
                .sent(false)
                .type(NotificationType.ACTIVITY)
                .createdAt(LocalDateTime.now())
                .subject(subject)
                .build();

        String body = emailTemplateService.buildActivityNotification(user, activity, message, title);

        try {
            notifyUser(user.getEmail(), subject, body);
            notification.setSent(true);

        } catch (Exception e) {
            System.out.println("Não foi possível de enviar o email " + e.getMessage());
        }

        notification = notificationRepository.save(notification);

        return notification;

    }

    public Notification createUserNotification(User user, String message, String subject, String title) {
        Notification notification = Notification.builder()
                .message(message)
                .user(user)
                .recipient(user.getEmail())
                .title(title)
                .sent(false)
                .type(NotificationType.USER)
                .createdAt(LocalDateTime.now())
                .subject(subject)
                .build();

        String body = emailTemplateService.buildUserNotification(user, message, title);
        try {
            notifyUser(user.getEmail(), subject, body);
            notification.setSent(true);

        } catch (Exception e) {
            System.out.println("Não foi possível de enviar o email " + e.getMessage());
        }

        notification = notificationRepository.save(notification);

        return notification;
    }

    public void notifyUser(String to, String subject, String body) {
        emailSender.send(to, subject, body);
    }

    public List<GetNotificationResponseDTO> getMyNotifications(String email, Pageable pageable) {
        User user = userRepository.findByEmail(email)
                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Usuário não encontrado."));

        Page<Notification> notifications = notificationRepository.findAllByUserIdOrderByCreatedAtDesc(user.getId(),
                pageable);

        return notifications.stream()
                .map(notification -> modelMapper.map(notification, GetNotificationResponseDTO.class))
                .toList();
    }

    public PathNotificationResponseDTO setMyNotificationAsRead(String email, Long notificationId) {
        userRepository.findByEmail(email)
                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Usuário não encontrado."));

        Notification notification = notificationRepository.findById(notificationId).orElseThrow(
                () -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Notificação não encontrada."));

        boolean notificationOfThisUser = notification.getUser().getEmail().equals(email);

        if (!notificationOfThisUser)
            throw new ResponseStatusException(HttpStatus.FORBIDDEN,
                    "Essa notificação não lhe pertence.");

        notification.setRead(true);
        notification.setReadAt(LocalDateTime.now());

        notificationRepository.save(notification);

        return modelMapper.map(notification, PathNotificationResponseDTO.class);
    }

    public GetHaveNotificationNotReadResponseDTO getHaveNotificationNotRead(String email) {

        User user = userRepository.findByEmail(email)
                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Usuário não encontrado."));

        GetHaveNotificationNotReadResponseDTO getHaveNotificationNotReadResponse = new GetHaveNotificationNotReadResponseDTO(
                user.getNotifications().stream().anyMatch(notification -> !notification.isRead()));

        return getHaveNotificationNotReadResponse;
    }

    public Integer getTotalOfMyNotifications(String email) {
        User user = userRepository.findByEmail(email)
                .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Usuário não encontrado."));

        return user.getNotifications().size();
    }

}
