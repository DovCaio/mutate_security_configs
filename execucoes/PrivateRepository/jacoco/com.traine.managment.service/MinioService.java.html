package com.traine.managment.service;

import java.io.InputStream;
import java.util.Collection;
import java.util.UUID;

import org.modelmapper.ModelMapper;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;
import org.springframework.web.server.ResponseStatusException;

import com.traine.managment.dto.files.FileDownloadDTO;
import com.traine.managment.dto.files.FileUrlGetResponseDTO;
import com.traine.managment.dto.files.UploadPostResponseDTO;
import com.traine.managment.interfaces.files.FileStorageService;

import io.minio.GetObjectArgs;
import io.minio.GetObjectResponse;
import io.minio.GetPresignedObjectUrlArgs;
import io.minio.MinioClient;
import io.minio.PutObjectArgs;
import io.minio.RemoveObjectArgs;
import io.minio.StatObjectArgs;
import io.minio.StatObjectResponse;
import io.minio.http.Method;

@Service
public class MinioService implements FileStorageService {
    @Autowired
    private MinioClient minioClient;

    @Value("${minio.bucket}")
    private String bucket;

    private String generateFileKey(MultipartFile file) {

        String originalName = file.getOriginalFilename();
        if (originalName == null)
            originalName = "file";

        String extension = "";
        String baseName = originalName;

        if (originalName.contains(".")) {
            extension = originalName.substring(originalName.lastIndexOf("."));
            baseName = originalName.substring(0, originalName.lastIndexOf("."));
        }

        return UUID.randomUUID()
                + "__"
                + baseName
                + extension;
    }

    @Override
    public UploadPostResponseDTO upload(MultipartFile file) {
        try {

            String fileKey = generateFileKey(file);

            minioClient.putObject(
                    PutObjectArgs.builder()
                            .bucket(bucket)
                            .object(fileKey)
                            .stream(file.getInputStream(), file.getSize(), -1)
                            .contentType(file.getContentType())
                            .build());

            return new UploadPostResponseDTO(
                    fileKey);

        } catch (Exception e) {
            throw new ResponseStatusException(HttpStatus.SERVICE_UNAVAILABLE,
                    "Não foi possível criar esse arquivo.");
        }
    }

    @Override
    public FileUrlGetResponseDTO generateUrl(String fileKey) {

        if (!exists(fileKey))
            throw new ResponseStatusException(HttpStatus.NOT_FOUND,
                    "Não existe nenhum arquivo com essa chave.");

        String url;
        try {

            url = minioClient.getPresignedObjectUrl(
                    GetPresignedObjectUrlArgs.builder()
                            .bucket(bucket)
                            .object(fileKey)
                            .method(Method.GET)
                            .expiry(60 * 10) // 10 minutos
                            .build());

        } catch (Exception e) {
            throw new ResponseStatusException(HttpStatus.SERVICE_UNAVAILABLE,
                    "Não foi possível criar esse arquivo.");
        }
        return new FileUrlGetResponseDTO(url);

    }

    @Override
    public void delete(String fileKey) {

        if (!exists(fileKey))
            throw new ResponseStatusException(HttpStatus.NOT_FOUND,
                    "Não existe nenhum arquivo com essa chave.");

        try {

            minioClient.removeObject(
                    RemoveObjectArgs.builder()
                            .bucket(bucket)
                            .object(fileKey)
                            .build());

        } catch (Exception e) {
            throw new ResponseStatusException(HttpStatus.SERVICE_UNAVAILABLE,
                    "Não foi possível criar esse arquivo.");
        }
    }

    @Override
    public boolean exists(String fileKey) {
        try {
            minioClient.statObject(
                    StatObjectArgs.builder()
                            .bucket(bucket)
                            .object(fileKey)
                            .build());
            return true;
        } catch (Exception e) {
            return false;
        }
    }

    @Override
    public boolean allExists(Collection<String> filesKeys) {
        if (filesKeys == null || filesKeys.isEmpty()) {
            return true;
        }

        return filesKeys.stream()
                .allMatch(this::exists);
    }

    public FileDownloadDTO download(String fileKey) {
        try {

            if (!exists(fileKey)) {
                throw new ResponseStatusException(HttpStatus.NOT_FOUND,
                        "Arquivo não encontrado");
            }

            GetObjectResponse stream = minioClient.getObject(
                    GetObjectArgs.builder()
                            .bucket(bucket)
                            .object(fileKey)
                            .build());

            byte[] content = stream.readAllBytes();

            StatObjectResponse stat = minioClient.statObject(
                    StatObjectArgs.builder()
                            .bucket(bucket)
                            .object(fileKey)
                            .build());

            MediaType contentType = MediaType.parseMediaType(
                    stat.contentType());

            return new FileDownloadDTO(content, fileKey, contentType);

        } catch (Exception e) {
            throw new ResponseStatusException(HttpStatus.SERVICE_UNAVAILABLE,
                    "Erro ao baixar arquivo");
        }
    }

}
