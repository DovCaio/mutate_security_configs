package com.traine.managment.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.domain.Pageable;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PatchMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import org.springframework.security.core.Authentication;
import org.springframework.security.core.userdetails.UserDetails;

import com.traine.managment.dto.notification.GetHaveNotificationNotReadResponseDTO;
import com.traine.managment.dto.notification.GetNotificationResponseDTO;
import com.traine.managment.dto.notification.PathNotificationResponseDTO;
import com.traine.managment.service.NotificationService;

@RestController
@RequestMapping("/notification")
public class NotificationController {

    @Autowired
    NotificationService notificationService;

    @GetMapping("/my")
    public ResponseEntity<List<GetNotificationResponseDTO>> getMyNotifications(Authentication auth, Pageable pageable) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();
        return ResponseEntity.ok().body(notificationService.getMyNotifications(email, pageable));
    }

    @PatchMapping("/{id}/read")
    public ResponseEntity<PathNotificationResponseDTO> setMyNotificationAsRead(@PathVariable Long id,
            Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();
        return ResponseEntity.ok().body(notificationService.setMyNotificationAsRead(email, id));

    }

    @GetMapping("/haveSomeOneNotRead")
    public ResponseEntity<GetHaveNotificationNotReadResponseDTO> getHaveNotificationNotRead(Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();
        return ResponseEntity.ok().body(notificationService.getHaveNotificationNotRead(email));
    }

    @GetMapping("/my/total") // Tem que fazer testes para isso daqui
    public ResponseEntity<Integer> getTotalOfMyNotifications(Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();
        return ResponseEntity.ok().body(notificationService.getTotalOfMyNotifications(email));
    }

}
