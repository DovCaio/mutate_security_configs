package com.traine.managment.controller;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpHeaders;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.multipart.MultipartFile;

import com.traine.managment.dto.files.FileDownloadDTO;
import com.traine.managment.dto.files.FileUrlGetResponseDTO;
import com.traine.managment.dto.files.UploadPostResponseDTO;
import com.traine.managment.interfaces.files.FileStorageService;

@RestController
@RequestMapping("/files")
public class FilesController {

    @Autowired
    private FileStorageService fileService;

    @PostMapping("/upload")
    public ResponseEntity<UploadPostResponseDTO> upload(@RequestParam("file") MultipartFile file) {

        UploadPostResponseDTO response = fileService.upload(file);

        return ResponseEntity.ok().body(response);

    }

    @GetMapping("/url/{fileKey}")
    public ResponseEntity<FileUrlGetResponseDTO> getUrl(@PathVariable("fileKey") String fileKey) {

        FileUrlGetResponseDTO response = fileService.generateUrl(fileKey);

        return ResponseEntity.ok().body(response);

    }

    @GetMapping("/download/{fileKey}")
    public ResponseEntity<byte[]> download(@PathVariable String fileKey) {

        FileDownloadDTO file = fileService.download(fileKey);

        return ResponseEntity.ok()
                .header(HttpHeaders.CONTENT_DISPOSITION,
                        "attachment; filename=\"" + file.getFileName() + "\"")
                .contentType(file.getContentType())
                .body(file.getContent());
    }

    @GetMapping("/view/{fileKey}")
    public ResponseEntity<byte[]> view(@PathVariable String fileKey) {

        FileDownloadDTO file = fileService.download(fileKey);

        return ResponseEntity.ok()
                .contentType(file.getContentType())
                .body(file.getContent());
    }

    @DeleteMapping("/{fileKey}")
    public ResponseEntity<Void> delete(@PathVariable("fileKey") String fileKey) {

        fileService.delete(fileKey);

        return ResponseEntity.noContent().build();

    }

    @GetMapping("/exists/{fileKey}")
    public ResponseEntity<Boolean> exists(@PathVariable("fileKey") String fileKey) {
        return ResponseEntity.ok().body(fileService.exists(fileKey));
    }

}
