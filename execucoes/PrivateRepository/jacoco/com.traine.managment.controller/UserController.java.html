package com.traine.managment.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.HttpStatusCode;
import org.springframework.http.ResponseEntity;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PatchMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.server.ResponseStatusException;

import com.traine.managment.dto.activity.GetRecentActivityResponseDTO;
import com.traine.managment.dto.user.GetTraineeActivitiesSumaryResponseDTO;
import com.traine.managment.dto.user.GetTraineeActivitiesYearDTO;
import com.traine.managment.dto.user.GetValidatorsActivitiesSumaryResponseDTO;
import com.traine.managment.dto.user.ChangePasswordRequestDTO;
import com.traine.managment.dto.user.CreateUserRequestDTO;
import com.traine.managment.dto.user.CreateUserResponseDTO;
import com.traine.managment.dto.user.GetSafeIdentificationDTO;
import com.traine.managment.dto.user.MeUserResponseDTO;
import com.traine.managment.dto.user.PutUserRequestDTO;
import com.traine.managment.dto.user.PutUserResponseDTO;
import com.traine.managment.dto.user.UserGetResponseDTO;
import com.traine.managment.enums.user.Role;
import com.traine.managment.service.UserService;

import jakarta.validation.Valid;

import org.springframework.security.core.Authentication;
import org.springframework.security.core.userdetails.UserDetails;

@RestController
@RequestMapping("/users")
public class UserController {

    @Autowired
    private UserService service;

    @PostMapping
    @PreAuthorize("hasAnyRole('ADMIN', 'MANAGMENT')")
    public ResponseEntity<CreateUserResponseDTO> createManager(@RequestBody @Valid CreateUserRequestDTO request,
            Authentication auth) {

        if (auth.getAuthorities().stream().anyMatch(a -> a.getAuthority().equals("ROLE_MANAGMENT"))
                && (request.getRole() == Role.ADMIN || request.getRole() == Role.MANAGMENT)) {
            throw new ResponseStatusException(HttpStatus.FORBIDDEN,
                    "Um Gerente não pode criar usuarios com papeis iguais, nem superior ao dele.");
        }

        CreateUserResponseDTO response = service.createUser(request);
        return ResponseEntity.status(HttpStatus.CREATED).body(response);
    }

    @GetMapping("/trainees")
    @PreAuthorize("hasAnyRole('ACTIVITY_MAKER', 'MANAGMENT')")
    public ResponseEntity<List<UserGetResponseDTO>> getAllTrainees() {
        return ResponseEntity.ok(service.getAllTrainees());
    }

    @GetMapping("/validators")
    public ResponseEntity<List<UserGetResponseDTO>> getAllValidators() {
        return ResponseEntity.ok(service.getAllValidators());
    }

    @PutMapping("")
    public ResponseEntity<PutUserResponseDTO> updateUser(@RequestBody @Valid PutUserRequestDTO request,
            Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();

        PutUserResponseDTO response = service.updateUser(request, email);
        return ResponseEntity.ok(response);

    }

    @PatchMapping("/change-password")
    public ResponseEntity<Void> changePassword(@RequestBody @Valid ChangePasswordRequestDTO request,
            Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();

        service.changePassword(request, email);
        return ResponseEntity.ok().build();
    }

    @GetMapping("/trainee/activities-summary")
    @PreAuthorize("hasRole('TRAINEE')")
    public ResponseEntity<GetTraineeActivitiesSumaryResponseDTO> getTraineeActivitiesSummary(Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();

        GetTraineeActivitiesSumaryResponseDTO response = service.getTraineeActivitiesSummary(email);
        return ResponseEntity.ok(response);
    }

    @GetMapping("/trainee/activities-year-quantity") // Isso daqui é para ver a necessidade, do quanto precisa-se dos
                                                     // estagiários, portanto não é baseado em quando terminou a
                                                     // atividade
    @PreAuthorize("hasAnyRole('MANAGMENT', 'ACTIVITY_MAKER')")
    public ResponseEntity<GetTraineeActivitiesYearDTO[]> getTraineeActivitiesYearQuantity() {

        GetTraineeActivitiesYearDTO[] response = service.getTraineeActivitiesYearQuantity();
        return ResponseEntity.ok(response);
    }

    @GetMapping("/validator/activities-summary") // Esse daqui é o mais perigoso, tem que testar mais
    @PreAuthorize("hasAnyRole('MANAGMENT', 'ACTIVITY_MAKER')")
    public ResponseEntity<GetValidatorsActivitiesSumaryResponseDTO> getValidatorsActivitiesSummary(
            Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();

        GetValidatorsActivitiesSumaryResponseDTO response = service.getValidatorsActivitiesSummary(email);
        return ResponseEntity.ok(response);
    }

    @GetMapping("/recent-activities")
    public ResponseEntity<List<GetRecentActivityResponseDTO>> getRecentActivities(Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();

        List<GetRecentActivityResponseDTO> response = service.getRecentActivities(email);
        return ResponseEntity.ok(response);
    }

    @GetMapping("/safe_identification")
    public ResponseEntity<GetSafeIdentificationDTO> getSafeIdentication(Authentication auth) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();

        GetSafeIdentificationDTO response = service.getSafeIdentication(email);

        return ResponseEntity.ok(response);
    }

}
