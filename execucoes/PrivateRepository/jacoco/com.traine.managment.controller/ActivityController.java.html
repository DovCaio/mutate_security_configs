package com.traine.managment.controller;

import java.util.Set;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.HttpStatusCode;
import org.springframework.http.ResponseEntity;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PatchMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import jakarta.validation.Valid;

import com.traine.managment.dto.activity.*;
import com.traine.managment.enums.user.Role;
import com.traine.managment.model.User;
import com.traine.managment.service.ActivityService;

import org.springframework.security.core.GrantedAuthority;
import org.springframework.security.core.annotation.AuthenticationPrincipal;

@RestController
@RequestMapping("/activity")
public class ActivityController {

    @Autowired
    private ActivityService activityService;

    @PostMapping
    @PreAuthorize("hasAnyRole('MANAGMENT', 'ADMIN', 'ACTIVITY_MAKER')")
    public ResponseEntity<PostActivityResponseDTO> createAnActivity(@RequestBody @Valid PostActivityRequestDTO body) {
        PostActivityResponseDTO response = activityService
                .createAnActivityInsertIntoGoogleCalendarAndSendNotification(body);
        return ResponseEntity.status(HttpStatus.CREATED).body(response);
    }

    @PatchMapping("/{id}")
    @PreAuthorize("hasAnyRole('MANAGMENT', 'ADMIN', 'ACTIVITY_MAKER')")
    public ResponseEntity<PatchActivityResponseDTO> patchAnActivity(@RequestBody @Valid PatchActivityRequestDTO body,
            @PathVariable Long id) {
        PatchActivityResponseDTO response = activityService.putNoticationAndUpdateGoogleCalendarAndSendNotification(id,
                body);
        return ResponseEntity.status(HttpStatus.OK).body(response);
    }

    @DeleteMapping("/{id}")
    @PreAuthorize("hasAnyRole('MANAGMENT', 'ADMIN', 'ACTIVITY_MAKER')")
    public ResponseEntity<Void> deleteAnActivity(@PathVariable Long id) {
        activityService.delete(id);
        return ResponseEntity.status(HttpStatus.NO_CONTENT).build();
    }

    @GetMapping()
    @PreAuthorize("hasAnyRole('MANAGMENT', 'ADMIN')")
    public ResponseEntity<Set<GetActivityIParticipateResponseDTO>> getAll() {
        Set<GetActivityIParticipateResponseDTO> activities = activityService.getAll();
        return ResponseEntity.status(HttpStatus.OK).body(activities);
    }

    @GetMapping("/{id}")
    @PreAuthorize("hasAnyRole('MANAGMENT', 'ADMIN', 'ACTIVITY_MAKER', 'TRAINEE')")
    public ResponseEntity<GetActivityResponseDTO> getById(@PathVariable Long id,
            @AuthenticationPrincipal UserDetails user) {
        GetActivityResponseDTO activity = activityService.getById(user.getUsername(), id);
        return ResponseEntity.status(HttpStatus.OK).body(activity);
    }

    @GetMapping("/i-participate")
    public ResponseEntity<Set<GetActivityIParticipateResponseDTO>> getAllWichIParticipate(
            @AuthenticationPrincipal UserDetails user) {
        Set<GetActivityIParticipateResponseDTO> activities = activityService.getAllWichIParticipate(user.getUsername());
        return ResponseEntity.status(HttpStatus.OK).body(activities);
    }

    @PatchMapping("/{id}/in_progress")
    @PreAuthorize("hasRole('TRAINEE')")
    public ResponseEntity<PathActivityProgressResponseDTO> makeAnActivityInProgress(
            @AuthenticationPrincipal UserDetails user, @PathVariable Long id) {
        PathActivityProgressResponseDTO progress = activityService.pathInProgress(user.getUsername(), id);
        return ResponseEntity.status(HttpStatus.OK).body(progress);
    }

    @PatchMapping("/{id}/awating_evaluation")
    @PreAuthorize("hasRole('TRAINEE')")
    public ResponseEntity<PathActivityProgressResponseDTO> makeAnActivityAwaitEvaluation(
            @AuthenticationPrincipal UserDetails user, @PathVariable Long id) {
        PathActivityProgressResponseDTO progress = activityService.pathAwaitEvaluation(user.getUsername(), id);
        return ResponseEntity.status(HttpStatus.OK).body(progress);
    }

    @PatchMapping("/{id}/evaluate")
    @PreAuthorize("hasAnyRole('ACTIVITY_MAKER', 'MANAGMENT')")
    public ResponseEntity<PathActivityProgressResponseDTO> evaluateActivity(
            @AuthenticationPrincipal UserDetails user, @PathVariable Long id,
            @RequestBody @Valid PatchEvaluateRequestDTO body) {

        PathActivityProgressResponseDTO progress = activityService.evaluate(user.getUsername(), id, body);
        return ResponseEntity.status(HttpStatus.OK).body(progress);
    }

    @PostMapping("/{id}/aux-files")
    @PreAuthorize("hasAnyRole('ACTIVITY_MAKER', 'MANAGMENT')")
    public ResponseEntity<AddNewAuxFileResponse> addAuxFiles(@AuthenticationPrincipal UserDetails user,
            @PathVariable Long id, @RequestBody @Valid AddNewAuxFileRequest body) {
        AddNewAuxFileResponse response = activityService.addAuxFiles(id, user.getUsername(), body);
        return ResponseEntity.ok().body(response);
    }

    @DeleteMapping("/{id}/aux-files/{fileKey}")
    @PreAuthorize("hasAnyRole('ACTIVITY_MAKER', 'MANAGMENT')")
    public ResponseEntity<AddNewAuxFileResponse> deleteAuxFiles(@AuthenticationPrincipal UserDetails user,
            @PathVariable Long id, @PathVariable String fileKey) {
        activityService.deleteAuxFile(id, fileKey, user.getUsername());
        return ResponseEntity.noContent().build();
    }

    @PostMapping("/{id}/answer-files")
    @PreAuthorize("hasRole('TRAINEE')")
    public ResponseEntity<AddNewAnswerFileResponse> addAnswerFiles(@AuthenticationPrincipal UserDetails user,
            @PathVariable Long id, @RequestBody @Valid AddNewAnswerFileRequest body) {
        AddNewAnswerFileResponse response = activityService.addAnswerFiles(id, user.getUsername(), body);
        return ResponseEntity.ok().body(response);
    }

    @DeleteMapping("/{id}/answers-files/{fileKey}")
    @PreAuthorize("hasRole('TRAINEE')")
    public ResponseEntity<AddNewAuxFileResponse> deleteAnswerFiles(@AuthenticationPrincipal UserDetails user,
            @PathVariable Long id, @PathVariable String fileKey) {
        activityService.deleteAnswerFile(id, fileKey, user.getUsername());
        return ResponseEntity.noContent().build();
    }

}
