package com.traine.managment.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.ResponseEntity;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;

import com.traine.managment.dto.activityStoryLine.StoryLineCommentRequestDTO;
import com.traine.managment.dto.activityStoryLine.StoryLineCommentResponseDTO;
import com.traine.managment.dto.activityStoryLine.StoryLineGetResponseDTO;
import com.traine.managment.service.ActivityStoreLineService;

import org.springframework.web.bind.annotation.RequestBody;

@RestController
@RequestMapping("/activityStoryLine")
public class ActivityStoryLineController {

    @Autowired
    private ActivityStoreLineService activityStoreLineService;

    @GetMapping("/{activityId}")
    public ResponseEntity<Page<StoryLineGetResponseDTO>> getAll(@PathVariable("activityId") Long activityId,
            Authentication auth,
            Pageable pageable) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();

        Page<StoryLineGetResponseDTO> storyLinesPage = activityStoreLineService.getStoryLines(activityId, email,
                pageable);
        return ResponseEntity.ok().body(storyLinesPage);

    }

    @PostMapping("/{activityId}/comment")
    public ResponseEntity<StoryLineCommentResponseDTO> comment(@PathVariable("activityId") Long activityId,
            Authentication auth, @RequestBody StoryLineCommentRequestDTO body) {
        UserDetails userDetails = (UserDetails) auth.getPrincipal();
        String email = userDetails.getUsername();

        return ResponseEntity.ok().body(activityStoreLineService.makeAComment(activityId, email, body));
    }

}
