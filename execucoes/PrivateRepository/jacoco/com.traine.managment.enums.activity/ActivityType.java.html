package com.traine.managment.enums.activity;

public enum ActivityType {

    PRODUCAO_DE_MANUAL("Produção de manual"),
    PUBLICACAO_NO_SITE("Publicação no site"),
    POSTAGEM_NO_INSTAGRAM("Postagem no Instagram"),
    EDICAO_NO_SITE("Edição no site"),
    FORMATACAO_DE_ARQUIVO("Formatação de arquivo"),
    CRIACAO_DE_DASHBOARD("Criação de dashboard"),
    TAREFA_RECORRENTE("Tarefa recorrente"),
    PROGRAMACAO_DE_SISTEMAS("Programação de sistemas"),
    OUTROS("Outros");

    private final String displayName;

    ActivityType(String displayName) {
        this.displayName = displayName;
    }

    public String getDisplayName() {
        return displayName;
    }
}
