package com.traine.managment.config;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.boot.CommandLineRunner;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.crypto.password.PasswordEncoder;

import com.traine.managment.enums.user.Role;
import com.traine.managment.enums.user.UserSector;
import com.traine.managment.model.User;
import com.traine.managment.repository.UserRepository;

@Configuration
public class DataInitializer {

    @Value("${app.security.admin-password}")
    private String adminPassword;

    private boolean prod = false;

    @Bean
    CommandLineRunner init(UserRepository repository, PasswordEncoder passwordEncoder) {
        return args -> {

            if (repository.existsByEmail("admin@admin.com")) {
                return;
            }

            User user = new User();
            user.setEmail("admin@admin.com");
            user.setName("Administrador");
            user.setPassword(passwordEncoder.encode(adminPassword));
            user.setRole(Role.ADMIN);
            user.setSector(UserSector.PRE);

            repository.save(user);

            System.out.println("Usuário admin criado!");

            if (!prod) {
                User user2 = new User();
                user2.setEmail("erica@erica.com");
                user2.setName("Erica");
                user2.setPassword(passwordEncoder.encode("12345678"));
                user2.setSafeIdentification("12345667789");
                user2.setRole(Role.MANAGMENT);
                user2.setSector(UserSector.PRE);
                User user3 = new User();
                user3.setEmail("caio@caio.com");
                user3.setName("Caio");
                user3.setPassword(passwordEncoder.encode("12345678"));
                user3.setSafeIdentification("6543321212");
                user3.setRole(Role.TRAINEE);
                user3.setSector(UserSector.PRE);

                repository.save(user2);
                repository.save(user3);
            }
        };
    }
}
