package com.traine.managment.model;

import java.time.LocalDate;
import java.util.ArrayList;
import java.util.Collection;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

import com.traine.managment.enums.activity.ActivityStatus;
import com.traine.managment.enums.activity.ActivityType;
import com.traine.managment.dto.activity.PatchActivityRequestDTO;
import com.traine.managment.enums.activity.ActivityPriority;

import jakarta.persistence.CascadeType;
import jakarta.persistence.CollectionTable;
import jakarta.persistence.Column;
import jakarta.persistence.ElementCollection;
import jakarta.persistence.Entity;
import jakarta.persistence.EnumType;
import jakarta.persistence.Enumerated;
import jakarta.persistence.GeneratedValue;
import jakarta.persistence.Id;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.JoinTable;
import jakarta.persistence.ManyToMany;
import jakarta.persistence.OneToMany;
import jakarta.persistence.Table;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

@Entity
@Getter
@Setter
@Builder
@AllArgsConstructor
@NoArgsConstructor
public class Activity {

    @Id
    @GeneratedValue
    @Column(name = "id_pk")
    private Long id;

    @Column(name = "description")
    private String description;

    @Column(name = "activity_type")
    @Enumerated(EnumType.STRING)
    private ActivityType activityType;

    @ManyToMany
    @JoinTable(name = "activity_trainees", joinColumns = @JoinColumn(name = "activity_id"), inverseJoinColumns = @JoinColumn(name = "user_id"))
    private Set<User> trainees;

    @ManyToMany
    @JoinTable(name = "activity_validators", joinColumns = @JoinColumn(name = "activity_id"), inverseJoinColumns = @JoinColumn(name = "user_id"))
    private Set<User> validators;

    @Column(name = "term")
    private LocalDate term;

    @Column(name = "createAt")
    private LocalDate createdAt;

    @Column(name = "finished_at")
    private LocalDate finishedAt;

    @Column(name = "status")
    @Enumerated(EnumType.STRING)
    private ActivityStatus activityStatus;

    @Column(name = "priority")
    @Enumerated(EnumType.STRING)
    private ActivityPriority activityPriority;

    @OneToMany(mappedBy = "activity")
    private Set<GoogleCalendarEvent> googleEvent;

    @OneToMany(mappedBy = "activity")
    private Set<TrelloActivityCard> trelloCardId;

    @ElementCollection
    @CollectionTable(name = "activity_auxiliary_files", joinColumns = @JoinColumn(name = "activity_id"))
    @Column(name = "aux_file_keys")
    @Builder.Default
    private Set<String> auxiliaryFilesKeys = new HashSet<>();

    @ElementCollection
    @CollectionTable(name = "activity_answer_files", joinColumns = @JoinColumn(name = "activity_id"))
    @Column(name = "answer_file_keys")
    @Builder.Default
    private Set<String> answerFilesKeys = new HashSet<>();

    @OneToMany(mappedBy = "activity")
    @Builder.Default
    private List<ActivityStoryLine> storyLines = new ArrayList<>();

    public boolean haveAuxiliarKey(String fileKey) {
        return auxiliaryFilesKeys.contains(fileKey);
    }

    public void addAuxiliarFileKey(String key) {

        if (key != null
                && !key.isBlank()
                && !auxiliaryFilesKeys.contains(key)) {

            auxiliaryFilesKeys.add(key);
        }
    }

    public void addAuxiliariesFileKeys(Collection<String> keys) {

        if (keys != null && !keys.isEmpty()) {

            keys.stream()
                    .forEach(this::addAuxiliarFileKey);
        }
    }

    public void deleteAuxiliarFileKey(String key) {
        if (key != null
                && !key.isBlank()
                && auxiliaryFilesKeys.contains(key))
            auxiliaryFilesKeys.remove(key);
    }

    public void delteAuxiliariesFilesKeys(Collection<String> keys) {
        if (keys != null && !keys.isEmpty())
            keys.stream().forEach(this::deleteAuxiliarFileKey);
    }

    public boolean haveAnswerKey(String fileKey) {
        return answerFilesKeys.contains(fileKey);
    }

    public void addAnwerFileKey(String key) {

        if (key != null
                && !key.isBlank()
                && !auxiliaryFilesKeys.contains(key)) {

            answerFilesKeys.add(key);
        }
    }

    public void addAnswerFileKeys(Collection<String> keys) {

        if (keys != null && !keys.isEmpty()) {

            keys.stream()
                    .forEach(this::addAnwerFileKey);
        }
    }

    public void finishedAt() {
        this.finishedAt = LocalDate.now();
    }

    public void deleteAnswerFileKey(String key) {
        if (key != null
                && !key.isBlank()
                && answerFilesKeys.contains(key))
            answerFilesKeys.remove(key);
    }

    public void deleteAnswerFilesKeys(Collection<String> keys) {
        if (keys != null && !keys.isEmpty())
            keys.stream().forEach(this::deleteAnswerFileKey);
    }

    public Set<User> getParticipants() {
        Set<User> participants = new HashSet<>(this.getTrainees());
        participants.addAll(this.getValidators());
        return participants;
    }

    public boolean areValidator(User user) {
        return validators.stream().anyMatch(validator -> validator.getId().equals(user.getId()));
    }

    public boolean areTrainee(User user) {
        return trainees.stream().anyMatch(trainee -> trainee.getId().equals(user.getId()));
    }

    public boolean areParticipant(User user) {
        return getParticipants().stream().anyMatch(participant -> participant.getId().equals(user.getId()));
    }

    public String differenceBetweenPutDto(PatchActivityRequestDTO dto) {
        List<String> changes = new ArrayList<>();

        if (dto.getDescription() != null && !dto.getDescription().equals(this.description)) {
            changes.add(String.format("Descrição alterada de '%s' para '%s'", this.description, dto.getDescription()));
        }

        if (dto.getActivityType() != null && !dto.getActivityType().equals(this.activityType)) {
            changes.add(String.format("Tipo alterado de %s para %s", this.activityType, dto.getActivityType()));
        }

        if (dto.getActivityPriority() != null && !dto.getActivityPriority().equals(this.activityPriority)) {
            changes.add(String.format("Prioridade alterada de %s para %s", this.activityPriority,
                    dto.getActivityPriority()));
        }

        if (dto.getTerm() != null && !dto.getTerm().equals(this.term)) {
            changes.add(String.format("Prazo alterado de %s para %s", this.term, dto.getTerm()));
        }

        if (changes.isEmpty()) {
            return "Nenhuma alteração detectada nos campos principais.";
        }

        return String.join("\n", changes);
    }
}
